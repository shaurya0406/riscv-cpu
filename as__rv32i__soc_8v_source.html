493module main_memory #(parameter MEMORY_DEPTH=1024) ( //Instruction and Data memory (combined memory)
494 input wire i_clk,

533 if(i_wr_mask[3]) memory_regfile[i_data_addr[$clog2(MEMORY_DEPTH)-1:2]][31:24] <= i_data_in[31:24];
534 end

536 end

541module uart #( //UART (TX only)
542 parameter CLOCK_FREQ = 12_000_000,//Input clock frequency
543 parameter BAUD_RATE = 9600, //UART Baud rate
544 parameter UART_TX_DATA = 8140, //memory-mapped address for TX (write to UART)
545 parameter UART_TX_BUSY = 8144, //memory-mapped address to check if TX is busy (has ongoing request)
546 parameter UART_RX_BUFFER_FULL = 8148, //memory-mapped address to check if a read has completed
547 parameter UART_RX_DATA = 8152, //memory-mapped address for RX (read the data)
548 parameter DBIT = 8, //UART Data Bits
549 parameter SBIT = 1 //UART Stop Bits
550 )(
551 input wire clk,
552 input wire rst_n,
553 input wire[31:0] uart_rw_address, //read/write address (access the memory-mapped registers for controlling UART)
554 input wire[DBIT - 1:0 ] uart_wdata, //TX data
555 input wire uart_wr_en, //write-enable
556 input wire uart_rx, //UART RX line
557 output wire uart_tx, //UART TX line
558 output reg[DBIT - 1:0] uart_rdata, //data read from memory-mapped register
559 input wire i_stb_data, //request to access UART
560 output reg o_ack_data //acknowledge by UART
561 );
562
563
564 localparam DVSR = CLOCK_FREQ/(16*BAUD_RATE);
565 localparam DVSR_WIDTH = $clog2(DVSR); //array size needed by DVSR
566 localparam SB_TICK = 16*SBIT;
567
568 //FSM state declarations
569 localparam[1:0] idle=2'd0,
570 start=2'd1,
571 data=2'd2,
572 stop=2'd3;
573
574 reg[DBIT - 1:0] uart_busy;
575 reg tx_done_tick;
576 reg[1:0] state_reg,state_nxt;
577 reg[3:0] s_reg,s_nxt; //count to 16 for every data bit
578 reg[2:0] n_reg,n_nxt; //count the number of data bits already transmitted
579 reg[DBIT - 1:0] din_reg,din_nxt; //stores the word to be transmitted
580 reg tx_reg,tx_nxt;
581 reg s_tick;
582 reg wr_uart;
583 reg[1:0] state_reg_rx,state_nxt_rx;
584 reg[3:0] s_reg_rx,s_nxt_rx; //check if number of ticks is 7(middle of start bit), or 15(middle of a data bit)
585 reg[2:0] n_reg_rx,n_nxt_rx; //checks how many data bits is already passed(value is 7 for last bit)
586 reg[7:0] b_reg,b_nxt; //stores 8-bit binary value of received data bits
587 reg[7:0] dout; //data read from UART
588 reg rx_done_tick; //goes high if a read is done
589 reg rx_buffer_full; //goes high if a read is done
590
591
592 //baud tick generator
593 reg[DVSR_WIDTH-1:0] counter=0;
594 always @(posedge clk,negedge rst_n) begin
595 if(!rst_n) counter<=0;
596 else begin
597 s_tick=0;
598 if(counter == DVSR-1) begin
599 s_tick=1;
600 counter<=0;
601 end
602 else begin
603 counter<=counter+1;
604 end
605
606 end
607 end
608 //Read memory-mapped registers
609 always @(posedge clk, negedge rst_n) begin
610 if(!rst_n) begin
611 uart_rdata <= 0;
612 o_ack_data <= 0;
613 end
614 else begin
615 if(i_stb_data && !uart_wr_en && uart_rw_address == UART_TX_BUSY) begin //read request to UART_TX_BUSY_ADDR (check if there is an ongoing request)
616 uart_rdata <= uart_busy;
617 end
618 else if(i_stb_data && !uart_wr_en && uart_rw_address == UART_RX_BUFFER_FULL) begin //read request to UART_RX_BUFFER_FULL (check if a read is completed)
619 uart_rdata <= rx_buffer_full;
620 end
621 else if(i_stb_data && !uart_wr_en && uart_rw_address == UART_RX_DATA) begin //read request to UART_RX_DATA (read the data)
622 uart_rdata <= dout;
623 end
624 o_ack_data <= i_stb_data;
625 end
626 end
627
628
629 /******************************** UART TX ****************************************/
630
631
632 //FSM register operation
633 always @(posedge clk,negedge rst_n) begin
634 if(!rst_n) begin
635 state_reg<=idle;
636 s_reg<=0;
637 n_reg<=0;
638 din_reg<=0;
639 tx_reg<=0;
640 end
641 else begin
642 state_reg<=state_nxt;
643 s_reg<=s_nxt;
644 n_reg<=n_nxt;
645 din_reg<=din_nxt;
646 tx_reg<=tx_nxt;
647 end
648 end
649
650 //FSM next-state logic
651 always @* begin
652 state_nxt=state_reg;
653 s_nxt=s_reg;
654 n_nxt=n_reg;
655 din_nxt=din_reg;
656 tx_nxt=tx_reg;
657 tx_done_tick=0;
658 uart_busy= 1; //uart is busy unless its in idle state
659 case(state_reg)
660 idle: begin
661 tx_nxt=1;
662 uart_busy = 0;
663 //start transmit operation when there is a write request to UART_TX_DATA_ADDR and we are in idle
664 if(uart_wr_en && i_stb_data && uart_rw_address == UART_TX_DATA && !uart_busy) begin
665 din_nxt=uart_wdata;
666 s_nxt=0;
667 state_nxt=start;
668 uart_busy = 1;
669 end
670 end
671 start: begin //wait to finish the start bit
672 tx_nxt=0;
673 if(s_tick==1) begin
674 if(s_reg==15) begin
675 s_nxt=0;
676 n_nxt=0;
677 state_nxt=data;
678 end
679 else s_nxt=s_reg+1;
680 end
681 end
682 data: begin //wait for all data bits to be transmitted serially
683 tx_nxt=din_reg[0];
684 if(s_tick==1) begin
685 if(s_reg==15) begin
686 din_nxt=din_reg>>1;
687 s_nxt=0;
688 if(n_reg==DBIT-1) state_nxt=stop;
689 else n_nxt=n_reg+1;
690 end
691 else s_nxt=s_reg+1;
692 end
693 end
694 stop: begin //wait to finish the stop bit
695 tx_nxt=1;
696 if(s_tick==1) begin
697 if(s_reg==SB_TICK-1) begin
698 tx_done_tick=1;
699 state_nxt=idle;
700 end
701 else s_nxt=s_reg+1;
702 end
703 end
704 default: state_nxt=idle;
705 endcase
706 end
707 assign uart_tx=tx_reg;
708 /*********************************************************************************/
709
710 /******************************** UART RX ****************************************/
711
712 //FSM register operation
713 always @(posedge clk,negedge rst_n) begin
714 if(!rst_n) begin
715 state_reg_rx<=idle;
716 s_reg_rx<=0;
717 n_reg_rx<=0;
718 b_reg<=0;
719 dout<=0;
720 rx_buffer_full<=0;
721 end
722 else begin
723 state_reg_rx<=state_nxt_rx;
724 s_reg_rx<=s_nxt_rx;
725 n_reg_rx<=n_nxt_rx;
726 b_reg<=b_nxt;
727 if(rx_done_tick) begin
728 dout <= b_reg; //memory-mapped register storing the completed read data
729 rx_buffer_full <= 1'b1; //memory-mapped register to check if a read is done
730 end
731 else if(i_stb_data && !uart_wr_en && uart_rw_address == UART_RX_DATA) begin //read request to UART_RX_DATA (read the data)
732 rx_buffer_full <= 1'b0;
733 end
734 end
735 end
736
737 //FSM next-state logic
738 always @* begin
739 state_nxt_rx=state_reg_rx;
740 s_nxt_rx=s_reg_rx;
741 n_nxt_rx=n_reg_rx;
742 b_nxt=b_reg;
743 rx_done_tick=0;
744 case(state_reg_rx)
745 idle: if(uart_rx==0) begin //wait for start bit(rx of zero)
746 s_nxt_rx=0;
747 state_nxt_rx=start;
748 end
749 start: if(s_tick==1) begin //wait for middle of start bit
750 if(s_reg_rx==7) begin
751 s_nxt_rx=0;
752 n_nxt_rx=0;
753 state_nxt_rx=data;
754 end
755 else s_nxt_rx=s_reg_rx+1;
756 end
757 data: if(s_tick==1) begin //wait to pass all middle points of every data bits
758 if(s_reg_rx==15) begin
759 b_nxt={uart_rx,b_reg[7:1]};
760 s_nxt_rx=0;
761 if(n_reg_rx==DBIT-1) state_nxt_rx=stop;
762 else n_nxt_rx=n_reg_rx+1;
763 end
764 else s_nxt_rx=s_reg_rx+1;
765 end
766 stop: if(s_tick==1) begin //wait to pass the required stop bits
767 if(s_reg_rx==SB_TICK-1) begin
768 rx_done_tick=1;
769 state_nxt_rx=idle;
770 end
771 else s_nxt_rx=s_reg_rx+1;
772 end
773 default: state_nxt_rx=idle;
774 endcase
775 end
776 /*********************************************************************************/
777
778endmodule

782module i2c //SCCB mode(no pullups resistors needed) [REPEATED START NOT SUPPORTED]
783 #(parameter main_clock=12_000_000, //frequency of clk
784 freq=100_000, //i2c freqeuncy
785 addr_bytes=2,//addr_bytes=number of bytes of an address
786 I2C_START=8100, //write-only memory-mapped address to start i2c (write the i2c slave address)
787 I2C_WRITE=8104, //write-only memory-mapped address for sending data to slave
788 I2C_READ=8108, //read-only memory-mapped address to read data received from slave (this will also continue reading from slave)
789 I2C_BUSY=8112, //read-only memory-mapped address to check if i2c is busy (cannot accept request)
790 I2C_ACK=8116, //read-only memory-mapped address to check if last access has benn acknowledge by slave
791 I2C_READ_DATA_READY=8120, //read-only memory-mapped address to check if data to be received from slave is ready
792 I2C_STOP=8124 //write-only memory-mapped address to stop i2c (this is persistent thus must be manually turned off after stopping i2c)
793 )
794 (
795 input wire clk,
796 input wire rst_n,
797 input wire [31:0] i2c_rw_address, //read/write address (access the memory-mapped registers for controlling i2c)
798 input wire [7:0 ] i2c_wdata, //data to be written to slave or to memory-mapped registers of i2c
799 output reg [7:0] i2c_rdata, //data retrieved from slave or from the memory-mapped registers of i2c
800 input wire i2c_wr_en, //write-enable
801 inout wire scl, sda, //i2c bidrectional clock and data line
802 output reg o_ack_data, //acknowledge by i2c
803 input wire i_stb_data //request to access i2c
804 );
805
806
807 //memory-mapped registers for controlling i2c
808 wire[7:0] i2c_busy = {7'b0, !((state_q == idle) || (state_q == stop_or_write) || (state_q == stop_or_read))}; //check if busy (busy unless we are on these states)
809 wire[7:0] i2c_read_data_ready = {7'b0, (state_q == stop_or_read)}; //check if data is ready to be read (data is ready ONLY WHEN we are already waiting for another read request!)
810 reg[7:0] i2c_ack; //check last access request has been acknowledged by slave
811 reg[7:0] i2c_stop; //write non-zero data here to stop current read/write transaction
812
813
814 wire start = i_stb_data;
815 wire[7:0] wr_data = i2c_wdata;
816 reg ack;
817 reg rd_tick;
818
819 localparam full= (main_clock)/(2*freq),
820 half= full/2,
821 counter_width=$clog2(full);
822
823 //FSM state declarations
824 localparam[3:0] idle=0,
825 starting=1,
826 packet=2,
827 ack_servant=3,
828 read=4,
829 ack_master=5,
830 stop_1=6,
831 stop_2=7,
832 stop_or_read = 8,
833 stop_or_write = 9;
834 reg[3:0] state_q=idle,state_d;
835 reg op_q=0,op_d;
836 reg[3:0] idx_q=0,idx_d;
837 reg[8:0] wr_data_q=0,wr_data_d;
838 reg[7:0] rd_data_q,rd_data_d;
839 reg scl_q=0,scl_d;
840 reg sda_q=0,sda_d;
841 reg[counter_width-1:0] counter_q=0,counter_d;
842 reg[1:0] addr_bytes_q=0,addr_bytes_d;
843 wire scl_lo,scl_hi;
844 wire sda_in, sda_out;
845
846 //access memory-mapped register
847 always @(posedge clk, negedge rst_n) begin
848 if(!rst_n) begin
849 i2c_stop <= 0;
850 o_ack_data <= 0;
851 end
852 else begin
853 if(i_stb_data && i2c_wr_en && i2c_rw_address == I2C_STOP) i2c_stop <= i2c_wdata; //write to i2c_stop to stop transaction
854 if(i_stb_data && !i2c_wr_en && i2c_rw_address == I2C_ACK) i2c_rdata <= i2c_ack; //read i2c_ack to know if last access request has been ack by slave
855 if(i_stb_data && !i2c_wr_en && i2c_rw_address == I2C_READ_DATA_READY) i2c_rdata <= i2c_read_data_ready;//read this to know if data is ready to be read
856 if(i_stb_data && !i2c_wr_en && i2c_rw_address == I2C_BUSY) i2c_rdata <= i2c_busy; //read this to know if i2c is still busy
857 if(i_stb_data && !i2c_wr_en && i2c_rw_address == I2C_READ) i2c_rdata <= rd_data_q; //read this to know what has been read from slave (make sure I2C_READ_DATA_READY is already high)
858
859 o_ack_data <= i_stb_data;
860 end
861 end
862
863 //register operations
864 always@(posedge clk, negedge rst_n) begin
865 if(!rst_n) begin
866 state_q<=idle;
867 idx_q<=0;
868 wr_data_q<=0;
869 scl_q<=0;
870 sda_q<=0;
871 counter_q<=0;
872 rd_data_q<=0;
873 addr_bytes_q<=0;
874 i2c_ack <= 0;
875 end
876 else begin
877 state_q<=state_d;
878 op_q<=op_d;
879 idx_q<=idx_d;
880 wr_data_q<=wr_data_d;
881 scl_q<=scl_d;
882 sda_q<=sda_d;
883 if(i2c_busy[0]) counter_q<=counter_d; //freeze the scl (by freezing the counter) if we are on wait/idle state (not busy states)
884 rd_data_q<=rd_data_d;
885 addr_bytes_q<=addr_bytes_d;
886 i2c_ack <= {7'd0,ack};
887 end
888 end
889
890
891 //free-running clk, freq depends on parameter "freq"
892 always @* begin
893 counter_d=counter_q+1;
894 scl_d=scl_q;
895 if(state_q==idle || state_q==starting) scl_d=1'b1;
896 else if(counter_q==full[counter_width-1:0]) begin
897 counter_d=0;
898 scl_d=(scl_q==0)?1'b1:1'b0;
899 end
900 end
901
902 //I2C_START
903 //FSM next-state logic
904 always @* begin
905 state_d=state_q;
906 op_d=op_q;
907 idx_d=idx_q;
908 wr_data_d=wr_data_q;
909 rd_data_d=rd_data_q;
910 addr_bytes_d=addr_bytes_q;
911 sda_d=sda_q;
912 rd_tick=0;
913 ack=i2c_ack[0];
914 case(state_q)
915 idle: begin //wait for user to start i2c by writing the slave address to I2C_START
916 sda_d=1'b1;
917 addr_bytes_d=addr_bytes;
918 if(start==1'b1 && i2c_wr_en && i2c_rw_address == I2C_START) begin //wait for a request
919 wr_data_d={wr_data,1'b1}; //the last 1'b1 is for the ACK coming from the servant("1" means high impedance or "reading")
920 op_d= (wr_data[0])? 1:0; // if last bit(R/W bit) is one:read operation, else write operation
921 idx_d=8; //index to be used on transmitting the wr_data serially(MSB first)
922 state_d=starting;
923 end
924 end
925
926 starting: if(scl_hi) begin //start command, change sda to low while scl is high
927 sda_d=0;
928 state_d=packet;
929 end
930
931 packet: if(scl_lo) begin //transmit wr_data serially(MSB first)
932 sda_d= (wr_data_q[idx_q]==0)? 0:1'b1;
933 idx_d= idx_q-1;
934 if(idx_q==0) begin
935 state_d=ack_servant;
936 idx_d=0;
937 end
938 end
939
940 ack_servant: if(scl_hi) begin //wait for ACK bit response(9th bit) from servant
941 ack=!sda_in;
942 if(i2c_stop[0]) state_d=stop_1; //master can forcefully stops the transaction (i2c_stop is memory-mapped)
943 else if(op_q/* && addr_bytes_q==0*/) begin //start reading after writing "addr_bytes" of packets for address
944 idx_d=7;
945 state_d=read;
946 end
947 else begin //write next packet
948 state_d = stop_or_write;
949 idx_d=8;
950 end
951 end
952
953 stop_or_write: if(i2c_stop[0]) begin //wait until user explicitly say to either stop i2c or continue writing
954 state_d = stop_1;
955 end
956 else if(start && i2c_wr_en && i2c_rw_address == I2C_WRITE) begin//continue writing
957 state_d = packet;
958 wr_data_d={wr_data,1'b1};
959 addr_bytes_d=addr_bytes_q-1;
960 end
961
962 read: if(scl_hi) begin //read data from slave(MSB first)
963 rd_data_d[idx_q[2:0]]=sda_in;
964 idx_d=idx_q-1;
965 if(idx_q==0) state_d=ack_master;
966 end
967
968 ack_master: if(scl_lo) begin //master must ACK after receiving data from servant
969 sda_d=1'b0;
970 if(sda_q==0) begin //one whole bit(two scl_lo) had passed
971 rd_tick=1;
972 idx_d=7;
973 if(i2c_stop[0]) state_d=stop_1; //after receiving data, master can opt to stop
974 else state_d=stop_or_read;
975 end
976 end
977 stop_or_read: if(i2c_stop[0]) begin //wait until user explicitly say to either stop i2c or continue reading
978 state_d = stop_1;
979 end
980 else if(start && !i2c_wr_en && i2c_rw_address == I2C_READ) begin //continue reading when current data is read
981 state_d = read;
982 end
983
984 stop_1: if(scl_lo) begin
985 sda_d=1'b0;
986 state_d=stop_2;
987 end
988 stop_2: if(scl_hi) begin
989 sda_d=1'b1;
990 state_d=idle;
991 end
992 default: state_d=idle;
993 endcase
994 end
995
996 //i2c IO logic requires pull-ups (2 logic levels: 0 or Z)
997 //assign scl=scl_q? 1'bz:0; //bidiectional logic for pull-up scl
998 //assign sda=sda_q? 1'bz:0; //bidirectional logic for pull-up scl
999 //assign sda_in=sda;
1000
1001 //We have not used pull-ups here so logic can be 0 or 1 (instead of high impedance). This is similar to A protocol.
1002 wire is_reading;
1003 assign is_reading = (state_q==read || state_q==ack_servant);
1004 assign sda_out = sda_q;
1005
1006 //Vivado, use IOBUF primitive
1007 `ifndef ICARUS
1008 IOBUF sda_iobuf ( //Vivado IOBUF instantiation
1009 .IO(sda),
1010 .I(sda_out),//write SDA when is_reading low
1011 .T(is_reading),
1012 .O(sda_in) //read SDA when is_reading high
1013 );
1014 `endif
1015 //Icarus simulator
1016 `ifdef ICARUS
1017 assign sda = sda_q;
1018 `endif
1019
1020 assign scl = scl_q;
1021 assign scl_hi= scl_q==1'b1 && counter_q==half[counter_width-1:0] /*&& scl==1'b1*/; //scl is on the middle of a high(1) bit
1022 assign scl_lo= scl_q==1'b0 && counter_q==half[counter_width-1:0]; //scl is on the middle of a low(0) bit

1028module rv32i_clint #( //Core Logic Interrupt
1029 parameter CLK_FREQ_MHZ = 12, //input clock frequency in MHz
1030 // A MTIMER device has two separate base addresses: one for the MTIME register and another for the MTIMECMP registers.
1031 parameter MTIME_BASE_ADDRESS = 8008,
1032 MTIMECMP_BASE_ADDRESS = 8016,
1033 MSIP_BASE_ADDRESS = 8024
1034)(
1035 input wire clk,
1036 input wire rst_n,
1037 input wire[31:0] clint_address, //read/write address (access the memory-mapped registers for controlling i2c)
1038 input wire[31:0] clint_wdata, //data to be written to slave or to memory-mapped registers of i2c
1039 output reg[31:0] clint_rdata, //data retrieved from slave or from the memory-mapped registers of i2c
1040 input wire clint_wr_en, //write-enable
1041 input wire i_stb, //request to access CLINT
1042 output reg o_ack, //acknowledge by CLINT
1043 // Interrupts
1044 output wire o_timer_interrupt,
1045 output wire o_software_interrupt
1046);
1047 // This is based from RISC-V Advanced Core Local Interruptor
1048 // Specification: https://github.vom/riscv/riscv-aclint/blob/main/riscv-aclint.adoc
1049
1050 // This RISC-V ACLINT specification defines a set of memory mapped devices which provide
