21 *
22 * This module handles the read and write operations for a 32-integer base register file.
23 * It includes support for synchronous read and write operations, and it avoids writing to the

27`default_nettype none

29module as_rv32i_basereg
30 (
31 input wire i_clk, ///< Clock signal
32 input wire i_ce_read, ///< Clock enable for reading from basereg [STAGE 2]

42 reg[4:0] rs1_addr_q, rs2_addr_q;
43 reg[31:0] base_regfile[31:1]; ///< Base register file (with base_regfile[0] hardwired to zero)
44 wire write_to_basereg;
45
46 /**
47 * @brief Sequential logic for read and write operations.
48 *
49 * This block handles synchronous read and write operations based on clock signals.
50 * It reads source register addresses and data to be written, and it updates the base register file.
51 */
52 always @(posedge i_clk) begin
53 if(write_to_basereg) begin ///< Only write to register if stage 5 is previously enabled (output of stage 5[WRITEBACK] is registered so delayed by 1 clk)
54 base_regfile[i_rd_addr] <= i_rd; ///< Synchronous write
55 end
56 if(i_ce_read) begin ///< Only read the register if stage 2 is enabled [DECODE]
57 rs1_addr_q <= i_rs1_addr; ///< Synchronous read
58 rs2_addr_q <= i_rs2_addr; ///< Synchronous read
59 end
60 end
61
62 /**
63 * @brief Control signal for write operation.
64 *
65 * Determines whether a write operation should be performed on the base register file.
66 */
67 assign write_to_basereg = i_wr && i_rd_addr!=0;
68
69 /**
70 * @brief Output for source register 1 value.
71 *
72 * Determines the output value for source register 1 based on the read address.
73 */
74 assign o_rs1 = rs1_addr_q==0? 0: base_regfile[rs1_addr_q];
75
76 /**

78 *
79 * Determines the output value for source register 2 based on the read address.
80 */
81 assign o_rs2 = rs2_addr_q==0? 0: base_regfile[rs2_addr_q];
