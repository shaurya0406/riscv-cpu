12/*
13 During the pipeline's execute step, the as_rv32i_alu module acts as the RISC-V core's Arithmetic Logic Unit (ALU).
14 The ALU is in charge of carrying out arithmetic, logic, and comparison operations based on the instructions and operands supplied.
15 This module is an essential component of the RISC-V core since it processes instructions and computes the results needed for programme execution.
16 Among the functions are:
17
18 1. Operand Selection: Depending on the opcode, the module first selects the relevant operands for the ALU operation. The programme counter (PC) or the value of the first source register (rs1) can be operand A, whereas operand B can be either the second source register (rs2) or an instantaneous value.
19 2. ALU Operation: Depending on the instruction type, the ALU executes various operations such as ADD, SUB, SLT, SLTU, XOR, OR, AND, SLL, SRL, SRA, EQ, NEQ, GE, and GEU. The ALU operation's result is saved in the y_d register.
20 3. Branch and Jump Handling: The module computes the next PC value based on the kind of instruction (e.g., branch, jump, or jump-and-link). It also creates the o_change_pc signal, which indicates whether or not the PC needs to jump to a different address.
21 4. Register Writeback: Based on the instruction type, the module computes the value to be written back to the destination register (rd) and sets the relevant control signals (o_wr_rd and o_rd_valid). For example, it prevents branch or store instructions from writing to the destination register.
22 5. Stalling and Flushing: The ALU manages pipeline stalling and flushing. Because accessing data memory may require numerous cycles, it generates the o_stall_from_alu signal to stall the memory-access stage for load/store instructions. Based on the input signals, it also handles pipeline stalls and flushes (i_stall, i_force_stall, and i_flush).
23*/

37module as_rv32i_alu(
38 input wire i_clk,i_rst_n,
39 input wire[`ALU_WIDTH-1:0] i_alu, // ALU operation type from previous stage
40 input wire[4:0] i_rs1_addr, // Address for register source 1
41 output reg[4:0] o_rs1_addr, // Address for register source 1
42 input wire[31:0] i_rs1, // Source register 1 value
43 output reg[31:0] o_rs1, // Source register 1 value
44 input wire[31:0] i_rs2, // Source register 2 value
45 output reg[31:0] o_rs2, // Source register 2 value
46 input wire[31:0] i_imm, // Immediate value from previous stage
47 output reg[11:0] o_imm, // Immediate value
48 input wire[2:0] i_funct3, // Function type from previous stage
49 output reg[2:0] o_funct3, // Function type
50 input wire[`OPCODE_WIDTH-1:0] i_opcode, // Opcode type from previous stage
51 output reg[`OPCODE_WIDTH-1:0] o_opcode, // Opcode type
52 input wire[`EXCEPTION_WIDTH-1:0] i_exception, // Exception from decoder stage
53 output reg[`EXCEPTION_WIDTH-1:0] o_exception, // Exception: illegal inst,ecall,ebreak,mret
54 output reg[31:0] o_y, // Result of arithmetic operation
55
56 /// PC Control ///
57 input wire[31:0] i_pc, // Program Counter
58 output reg[31:0] o_pc, // pc register in pipeline
59 output reg[31:0] o_next_pc, // New pc value
60 output reg o_change_pc, // High if pc needs to branch
61
62 /// Basereg Control ///
63 output reg o_wr_rd, // Write rd to the base reg if enabled
64 input wire[4:0] i_rd_addr, // Address for destination register (from previous stage)
65 output reg[4:0] o_rd_addr, // Address for destination register
66 output reg[31:0] o_rd, // Value to be written back to destination register
67 output reg o_rd_valid, // High if o_rd is valid (not load nor csr instruction)
68
69 /// Pipeline Control ///
70 output reg o_stall_from_alu, // Prepare to stall next stage(memory-access stage) for load/store instruction
71 input wire i_ce, // Input clk enable for pipeline stalling of this stage
72 output reg o_ce, // Output clk enable for pipeline stalling of next stage
73 input wire i_stall, // Informs this stage to stall

81 wire alu_sub = i_alu[`SUB];
82 wire alu_slt = i_alu[`SLT];
83 wire alu_sltu = i_alu[`SLTU];
84 wire alu_xor = i_alu[`XOR];
85 wire alu_or = i_alu[`OR];
86 wire alu_and = i_alu[`AND];
87 wire alu_sll = i_alu[`SLL];
88 wire alu_srl = i_alu[`SRL];
89 wire alu_sra = i_alu[`SRA];
90 wire alu_eq = i_alu[`EQ];
91 wire alu_neq = i_alu[`NEQ];
92 wire alu_ge = i_alu[`GE];
93 wire alu_geu = i_alu[`GEU];
94 wire opcode_rtype = i_opcode[`RTYPE];
95 wire opcode_itype = i_opcode[`ITYPE];
96 wire opcode_load = i_opcode[`LOAD];
97 wire opcode_store = i_opcode[`STORE];
98 wire opcode_branch = i_opcode[`BRANCH];
99 wire opcode_jal = i_opcode[`JAL];
100 wire opcode_jalr = i_opcode[`JALR];
101 wire opcode_lui = i_opcode[`LUI];

103 wire opcode_system = i_opcode[`SYSTEM];

106 // ALU Internal Registers
107 reg[31:0] a; // Operand A
108 reg[31:0] b; // Operand B
109 reg[31:0] y_d; // ALU output
110 reg[31:0] rd_d; // Next value to be written back to destination register
111 reg wr_rd_d; // Write rd to basereg if enabled
112 reg rd_valid_d; // High if rd is valid (not load nor csr instruction)
113 reg[31:0] a_pc;
114 wire[31:0] sum;
115 wire stall_bit = o_stall || i_stall;
116
117 // Register the output of i_alu
118 always @(posedge i_clk, negedge i_rst_n) begin
119 if(!i_rst_n) begin
120 o_exception <= 0;
121 o_ce <= 0;
122 o_stall_from_alu <= 0;
123 end
124 else begin
125 if(i_ce && !stall_bit) begin // Update register only if this stage is enabled
126 o_opcode <= i_opcode;
127 o_exception <= i_exception;
128 o_y <= y_d;
129 o_rs1_addr <= i_rs1_addr;
130 o_rs1 <= i_rs1;
131 o_rs2 <= i_rs2;
132 o_rd_addr <= i_rd_addr;
133 o_imm <= i_imm[11:0];
134 o_funct3 <= i_funct3;
135 o_rd <= rd_d;
136 o_rd_valid <= rd_valid_d;
137 o_wr_rd <= wr_rd_d;
138 o_stall_from_alu <= i_opcode[`STORE] || i_opcode[`LOAD]; // Stall next stage(4th memory-access stage) when need to store/load since accessing data memory usually takes more than 1 cycle
139 o_pc <= i_pc;
140 end
141 if(i_flush && !stall_bit) begin // Flush this stage so clock-enable of next stage is disabled at next clock cycle
142 o_ce <= 0;
143 end
144 else if(!stall_bit) begin // Clock-enable will change only when not stalled
145 o_ce <= i_ce;
146 end
147 else if(stall_bit && !i_stall) o_ce <= 0; // If this stage is stalled but the next stage is not, disable the following stage's clock enablement at the next clock cycle (pipeline bubble).
148 end
149
150 end
151
152 // Determine operation used then compute for y (ALU output)
153 always @* begin
154 y_d = 0;
155
156 a = (opcode_jal || opcode_auipc)? i_pc:i_rs1; // a can either be pc or rs1
157 b = (opcode_rtype || opcode_branch)? i_rs2:i_imm; // b can either be rs2 or imm
158
159 if(alu_add) y_d = a + b;
160 if(alu_sub) y_d = a - b;
161 if(alu_slt || alu_sltu) begin
162 y_d = {31'b0, (a < b)};
163 if(alu_slt) y_d = (a[31] ^ b[31])? {31'b0,a[31]}:y_d;
164 end
165 if(alu_xor) y_d = a ^ b;
166 if(alu_or) y_d = a | b;
167 if(alu_and) y_d = a & b;
168 if(alu_sll) y_d = a << b[4:0];
169 if(alu_srl) y_d = a >> b[4:0];
170 if(alu_sra) y_d = $signed(a) >>> b[4:0];
171 if(alu_eq || alu_neq) begin
172 y_d = {31'b0, (a == b)};
173 if(alu_neq) y_d = {31'b0,!y_d[0]};
174 end
175 if(alu_ge || alu_geu) begin
176 y_d = {31'b0, (a >= b)};
177 if(alu_ge) y_d = (a[31] ^ b[31])? {31'b0, b[31]}:y_d;
178 end
179 end
180
181 assign sum = a_pc + i_imm; // Share adder for all addition operation for less resource utilization
182
183 // Determine o_rd to be saved to basereg and next value of PC
184 always @* begin
185 o_flush = i_flush; // Flush this stage along with the previous stages
186 rd_d = 0;
187 rd_valid_d = 0;
188 o_change_pc = 0;
189 o_next_pc = 0;
190 wr_rd_d = 0;
191 a_pc = i_pc;
192 if(!i_flush) begin
193 if(opcode_rtype || opcode_itype) rd_d = y_d;
194 if(opcode_branch && y_d[0]) begin
195 o_next_pc = sum; // Branch if value of ALU is 1(true)
196 o_change_pc = i_ce; // Change PC when ce of this stage is high (o_change_pc is valid)
197 o_flush = i_ce;
198 end
199 if(opcode_jal || opcode_jalr) begin
200 if(opcode_jalr) a_pc = i_rs1;
201 o_next_pc = sum; // Jump to new PC
202 o_change_pc = i_ce; // Change PC when ce of this stage is high (make o_change_pc valid)
203 o_flush = i_ce;
204 rd_d = i_pc + 4; // Register the next pc value to destination register
205 end
206 end
207 if(opcode_lui) rd_d = i_imm;
208 if(opcode_auipc) rd_d = sum;
209
210 if(opcode_branch || opcode_store || (opcode_system && i_funct3 == 0) || opcode_fence ) wr_rd_d = 0; // i_funct3 == 0 are the non-csr system instructions
211 else wr_rd_d = 1; // Always write to the destination reg except when instruction is BRANCH or STORE or SYSTEM(except CSR system instruction)
212
213 if(opcode_load || (opcode_system && i_funct3!=0)) rd_valid_d = 0; // Value of o_rd for load and CSR write is not yet available at this stage
214 else rd_valid_d = 1;
215
216 // Stall logic (stall when upper stages are stalled, when forced to stall, or when needs to flush previous stages but are still stalled)
217 o_stall = (i_stall || i_force_stall) && !i_flush; // Stall when alu needs wait time
218 end
