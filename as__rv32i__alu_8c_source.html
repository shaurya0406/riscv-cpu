164 always @* begin
165 y_d = 0;

167 a = (opcode_jal || opcode_auipc)? i_pc:i_rs1;
168 b = (opcode_rtype || opcode_branch)? i_rs2:i_imm;
169
170 if(alu_add) y_d = a + b;
171 if(alu_sub) y_d = a - b;
172 if(alu_slt || alu_sltu) begin
173 y_d = {31'b0, (a < b)};
174 if(alu_slt) y_d = (a[31] ^ b[31])? {31'b0,a[31]}:y_d;
175 end
176 if(alu_xor) y_d = a ^ b;
177 if(alu_or) y_d = a | b;
178 if(alu_and) y_d = a & b;
179 if(alu_sll) y_d = a << b[4:0];
180 if(alu_srl) y_d = a >> b[4:0];
181 if(alu_sra) y_d = $signed(a) >>> b[4:0];
182 if(alu_eq || alu_neq) begin
183 y_d = {31'b0, (a == b)};
184 if(alu_neq) y_d = {31'b0,!y_d[0]};
185 end
186 if(alu_ge || alu_geu) begin
187 y_d = {31'b0, (a >= b)};
188 if(alu_ge) y_d = (a[31] ^ b[31])? {31'b0, b[31]}:y_d;
189 end
190 end

192 assign sum = a_pc + i_imm;
193
195 always @* begin
196 o_flush = i_flush;
197 rd_d = 0;
198 rd_valid_d = 0;
199 o_change_pc = 0;
200 o_next_pc = 0;
201 wr_rd_d = 0;
202 a_pc = i_pc;
203 if(!i_flush) begin
204 if(opcode_rtype || opcode_itype) rd_d = y_d;
205 if(opcode_branch && y_d[0]) begin

207 o_change_pc = i_ce;
208 o_flush = i_ce;
209 end
210 if(opcode_jal || opcode_jalr) begin
211 if(opcode_jalr) a_pc = i_rs1;
212 o_next_pc = sum;

214 o_flush = i_ce;

216 end
217 end
218 if(opcode_lui) rd_d = i_imm;
219 if(opcode_auipc) rd_d = sum;
220
221 if(opcode_branch || opcode_store || (opcode_system && i_funct3 == 0) || opcode_fence ) wr_rd_d = 0;
222 else wr_rd_d = 1;
223
224 if(opcode_load || (opcode_system && i_funct3!=0)) rd_valid_d = 0;
225 else rd_valid_d = 1;
