186 /* Volume 1 pg. 15: Instructions are 32 bits in length and must be aligned on a four-byte boundary in memory.
187 An instruction-address-misaligned exception is generated on a taken branch or unconditional jump
188 if the target address is not four-byte aligned. This exception is reported on the branch or jump
189 instruction, not on the target instruction. No instruction-address-misaligned exception is generated
190 for a conditional branch that is not taken. */
191 if((opcode_branch && i_y[0]) || opcode_jal || opcode_jalr) begin // branch or jump to new instruction
192 new_pc = i_pc[1:0] + i_csr_index[1:0];
193 if(opcode_jalr) new_pc = i_rs1[1:0] + i_csr_index[1:0];
194 is_inst_addr_misaligned = (new_pc == 2'b00)? 1'b0:1'b1; //i_pc (instruction address) must always be four bytes aligned
